[docs]def start(window: _CursesWindow, args: _Namespace) -> None:
    """Start the user interface."""

    task = _TuiTask(
        "ui",
        1 / args.rate,
        _ChannelEnvironment(),
        max_iterations=args.iterations,
    )
    stop_sig = _asyncio.Event()
    return _run_handle_stop(stop_sig, task.run(window, stop_sig=stop_sig))




[docs]def tui_cmd(args: _Namespace) -> int:
    """Execute the tui command."""

    getattr(_curses, "wrapper")(start, args)
    return 0




[docs]def add_tui_cmd(parser: _ArgumentParser) -> _CommandFunction:
    """Add tui-command arguments to its parser."""

    parser.add_argument(
        "-i",
        "--iterations",
        type=int,
        default=0,
        help=(
            "maximum number of program iterations (if greater "
            "than zero, default: %(default)s)"
        ),
    )
    parser.add_argument(
        "-r",
        "--rate",
        type=float,
        default=60.0,
        help=(
            "frequency (in Hz) to run the interface "
            "(default: %(default)s Hz)"
        ),
    )
    return tui_cmd
