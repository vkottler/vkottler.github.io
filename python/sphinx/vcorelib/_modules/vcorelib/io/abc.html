[docs]    def to_file(
        self,
        path: _Pathlike = None,
        encoding: str = _DEFAULT_ENCODING,
        **kwargs,
    ) -> None:
        """Write this object to a file."""

        with _normalize(
            path if path is not None else self.default_location()
        ).open("w", encoding=encoding) as stream:
            self.to_stream(stream, **kwargs)
            stream.flush()
